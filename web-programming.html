Pertemuan 3: Pengenalan JavaScript

JavaScript
// Contoh fungsi interaktif
function hitungLuas() {
    let panjang = document.getElementById('panjang').value;
    let lebar = document.getElementById('lebar').value;
    let luas = panjang * lebar;
    
    document.getElementById('hasil').innerHTML = 
        `Luas persegi panjang adalah: ${luas}`;
}

// Contoh event listener
document.getElementById('hitung').addEventListener('click', hitungLuas);

Pertemuan 5: Mengontrol Alur Program

    case 'A':
        echo "Sangat Baik";
        break;
    case 'B':
        echo "Baik";
        break;
    case 'C':
        echo "Cukup";
        break;
    default:

}
